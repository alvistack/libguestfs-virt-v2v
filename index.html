Installation

#!/bin/bash

set -euxo pipefail

sudo apt-key adv --keyserver keyserver.ubuntu.com --recv 644E28D3
sudo apt-add-repository "deb [arch=amd64] https://alvistack.github.io/libguestfs-virt-v2v stable main"
sudo apt update
sudo apt -y install virt-v2v
